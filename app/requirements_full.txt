#      venv\Scripts\pip install torch --extra-index-url https://download.pytorch.org/whl/cu124
#
#   3. 安裝本檔所有依賴：
#      venv\Scripts\pip install -r requirements_full.txt
#
#   4. （可選）若要啟用 Qwen3-ASR 4-bit 量化：
#      venv\Scripts\pip install bitsandbytes
#
#   5. 以 editable 模式安裝魔改版 stream-translator-gpt：
#      venv\Scripts\pip install -e ..\stream-translator-gpt
#
#   6. 執行打包：
#      .\build_exe.ps1
